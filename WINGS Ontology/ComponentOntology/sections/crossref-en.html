            <https://w3id.org/wings/TestingDomain#SECOND-INPUTFILE> ;

        <http://www.wings-workflows.org/ontology/component.owl#hasOutput>
            <https://w3id.org/wings/TestingDomain#OUTPUTFILE> ;
